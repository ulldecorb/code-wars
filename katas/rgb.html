            function rgb(r, g, b) {
                let roundDown = (a) => a<0 ? 0 : a; 
                let roundUp = (a) => a>255 ? 255 : a; 
                [r,g,b] = [...arguments].map((a) => roundUp(a)).map((a) => roundDown(a));
                let hex = (a) => a.toString(16).length<2 ? 0 + a.toString(16) : a.toString(16);
                let lowerCaseHex =  hex(r) + hex(g) + hex(b);
                let upperCaseHex = lowerCaseHex.split("").map((a) => a.match(/[a-z]/) ? a.toUpperCase() : a).join("");
                return upperCaseHex;
            }
